4. Resolver — exemple d'API

// lib/plan/session-meta.ts
resolveSessionMeta('trail-race-xyz', catalog) → {
  label: 'Trail Race',
  color: '#6B7280',        // gris
  category: 'run',
  isRunning: true,         // car category === 'run'
  defaultIntensity: 3
}

resolveSessionMeta('footing', catalog) → {
  label: 'Endurance Fondamentale',
  color: '#4ADE80',        // venant de SESSION_TYPE_COLORS
  category: 'run',
  isRunning: true,
  defaultIntensity: 2
}

resolveSessionMeta('unknown-slug', catalog) → {
  label: 'unknown-slug',    // fallback no-crash
  color: '#6B7280',
  category: 'other',
  isRunning: false,
  defaultIntensity: 2
}
